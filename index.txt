            <pre id="scriptCode">const SHEET_NAME_REG = 'registros';
const SHEET_NAME_FUNC = 'funcionarios';

function doGet(e)  { return handleRequest(e); }
function doPost(e) { return handleRequest(e); }

function handleRequest(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const p  = e.parameter || {};
  const action = p.action || (e.postData ? JSON.parse(e.postData.contents).action : '');
  const body   = e.postData ? JSON.parse(e.postData.contents) : p;
  let result = {};
  try {
    if (action === 'getFuncionarios') {
      result = { data: lerSheet(ss, SHEET_NAME_FUNC) };
    } else if (action === 'getRegistros') {
      result = { data: lerSheet(ss, SHEET_NAME_REG) };
    } else if (action === 'setFuncionarios') {
      gravarSheet(ss, SHEET_NAME_FUNC, body.data);
      result = { ok: true };
    } else if (action === 'setRegistros') {
      gravarSheet(ss, SHEET_NAME_REG, body.data);
      result = { ok: true };
    } else if (action === 'ping') {
      result = { ok: true, msg: 'Conexao ok!' };
    } else {
      result = { erro: 'Acao desconhecida' };
    }
  } catch(err) {
    result = { erro: err.message };
  }
  return ContentService
    .createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}

function lerSheet(ss, nome) {
  let sheet = ss.getSheetByName(nome);
  if (!sheet) return [];
  const data = sheet.getDataRange().getValues();
  if (data.length < 2) return [];
  const headers = data[0];

function iniciais(nome) {

function dataBRtoISO(br) { const [d,m,a] = br.split('/'); return `${a}-${m}-${d}`; }
function dataISOtoBR(iso) { const [a,m,d] = iso.split('-'); return `${d}/${m}/${a}`; }

function statusHoje(funcId) {
  const hoje = new Date().toLocaleDateString('pt-BR');
  const regs = registros.filter(r => r.funcId === funcId && r.data === hoje);
  if (!regs.length) return 'livre';
  return regs.sort((a,b)=>a.timestamp-b.timestamp).slice(-1)[0].tipo === 'entrada' ? 'entrada' : 'saida';
}
function totalHoje(funcId) {
  const hoje = new Date().toLocaleDateString('pt-BR');
  return calcTotal(registros.filter(r => r.funcId === funcId && r.data === hoje));
}
function calcTotal(regs) {
  let total = 0, entrada = null;
  [...regs].sort((a,b)=>a.timestamp-b.timestamp).forEach(r => {
    if (r.tipo === 'entrada') entrada = r.timestamp;
    else if (r.tipo === 'saida' && entrada) { total += r.timestamp - entrada; entrada = null; }
  });
  if (entrada) total += Date.now() - entrada;
  return total;
}
function msToHM(ms) {
  const h = Math.floor(ms/3600000), m = Math.floor((ms%3600000)/60000);
  return `${String(h).padStart(2,'0')}h${String(m).padStart(2,'0')}`;
}

  if (!linhas.length) { toast('Adicione ao menos uma linha.','error'); return; }
  const novos = []; let erro = false;
  linhas.forEach(linha => {
    const rowId = linha.id.replace('mr-','');
    const tipo  = document.getElementById(`mt-${rowId}`)?.value;
    const dVal  = document.getElementById(`md-${rowId}`)?.value;
    const hVal  = document.getElementById(`mh-${rowId}`)?.value;
    if (!tipo || !dVal || !hVal) { erro = true; return; }
    novos.push({ id:'r'+Date.now()+Math.random(), funcId, tipo, data: dataISOtoBR(dVal), timestamp: new Date(`${dVal}T${hVal}:00`).getTime(), manual: true });
  });
  if (erro) { toast('Preencha data e horário em todas as linhas.','error'); return; }
  const diasNovos  = [...new Set(novos.map(r=>r.data))];
  const diasApagar = [...new Set([...manualDiasOriginais, ...diasNovos])];
  registros = registros.filter(r => !(r.funcId === funcId && diasApagar.includes(r.data)));
  registros.push(...novos);
  closeModal();
  toast('Registros salvos!');
  renderCards(); renderRegistro();
  await salvarTudo();
}

function excluirDia(funcId, data) {
  if (!confirm(`Excluir todos os registros de ${data}?`)) return;
  registros = registros.filter(r => !(r.funcId===funcId && r.data===data));
  renderCards(); renderRegistro();
  toast('Registros do dia excluídos.','error');
  salvarTudo();
}

}

async function salvarFuncionario() {
  const nome  = document.getElementById('inputNome')?.value.trim();
  const cargo = document.getElementById('inputCargo')?.value.trim();
  if (!nome) { toast('Informe o nome do funcionário.','error'); return; }
  if (editandoId) {
    const f = funcionarios.find(x => x.id === editandoId);
    f.nome = nome; f.cargo = cargo || 'Sem cargo';
    toast('Funcionário atualizado!');
  } else {
    funcionarios.push({ id: 'f'+Date.now(), nome, cargo: cargo || 'Sem cargo' });
    toast('Funcionário cadastrado!');
  }
  closeModal(); renderCards(); renderRelSidebar();
  if (funcSelecionado) renderRegistro();
  await salvarTudo();
}

async function excluirFuncionario(id) {
  const f = funcionarios.find(x => x.id === id);
  if (!confirm(`Remover ${f.nome} e todos os registros de ponto?`)) return;
  funcionarios = funcionarios.filter(x => x.id !== id);
  registros    = registros.filter(r => r.funcId !== id);
  funcSelecionado = null;
  renderCards(); renderRelSidebar();
  document.getElementById('registroContent').innerHTML = `<div class="empty-state"><p>Selecione um funcionário na aba <strong>Funcionários</strong>.</p></div>`;
  toast('Funcionário removido.','error');
  await salvarTudo();
}

function renderRelSidebar() {
  const list = document.getElementById('relFuncList');
  if (!funcionarios.length) { list.innerHTML='<p style="font-size:13px;color:var(--text-muted)">Nenhum funcionário.</p>'; return; }
  list.innerHTML = funcionarios.map(f => `
    <div class="rel-func-item ${relFuncSelecionado===f.id?'active':''}" onclick="selecionarRelFunc('${f.id}')">
      <div class="rel-func-avatar">${iniciais(f.nome)}</div>
      ${f.nome.split(' ')[0]}
    </div>`).join('');
}
function selecionarRelFunc(id) { relFuncSelecionado=id; renderRelSidebar(); renderRelContent(); }
function renderRelatorio() { renderRelSidebar(); if (relFuncSelecionado) renderRelContent(); }
function renderRelContent() {
  const content = document.getElementById('relContent');
  const f = funcionarios.find(x => x.id === relFuncSelecionado);
  if (!f) return;
  const regs = registros.filter(r => r.funcId === f.id);
  const porMes = {};
  regs.forEach(r => { const [,m,a]=r.data.split('/'); const k=`${m}/${a}`; if(!porMes[k]) porMes[k]=[]; porMes[k].push(r); });
  const meses = Object.keys(porMes).sort((a,b)=>{
    const [ma,aa]=a.split('/'), [mb,ab]=b.split('/');
    return new Date(`${ab}-${mb}-01`) - new Date(`${aa}-${ma}-01`);
  }).reverse();
  const totalMs = calcTotal(regs);
  const diasTrabalhados = new Set(regs.map(r=>r.data)).size;
  const mediaDia = diasTrabalhados ? totalMs/diasTrabalhados : 0;
  let mesRows = meses.map(mes => {
    const mrs = porMes[mes], ms = calcTotal(mrs), dias = new Set(mrs.map(r=>r.data)).size;
    return `<tr><td>${mes}</td><td>${dias} dia${dias!==1?'s':''}</td><td style="color:var(--green);font-weight:600">${msToHM(ms)}</td><td style="color:var(--text-muted)">${dias?msToHM(ms/dias):'–'}</td></tr>`;
  }).join('') || `<tr><td colspan="4"><div class="empty-state" style="padding:24px"><p>Nenhum registro.</p></div></td></tr>`;
  content.innerHTML = `
    <div style="display:flex;align-items:center;gap:12px;margin-bottom:24px;flex-wrap:wrap">
      <div class="reg-avatar" style="width:44px;height:44px;font-size:16px">${iniciais(f.nome)}</div>
      <div><div style="font-size:18px;font-weight:700">${f.nome}</div><div style="font-size:13px;color:var(--text-muted)">${f.cargo}</div></div>
      <button class="btn btn-yellow" style="margin-left:auto;font-size:13px;padding:8px 16px" onclick="exportarCSV('${f.id}')">⬇ Exportar CSV</button>
    </div>
    <div class="rel-stats">
      <div class="rel-stat"><div class="rel-stat-label">Total geral</div><div class="rel-stat-val" style="color:var(--green)">${msToHM(totalMs)}</div></div>
      <div class="rel-stat"><div class="rel-stat-label">Dias trabalhados</div><div class="rel-stat-val">${diasTrabalhados}</div></div>
      <div class="rel-stat"><div class="rel-stat-label">Média/dia</div><div class="rel-stat-val" style="color:var(--yellow)">${diasTrabalhados?msToHM(mediaDia):'–'}</div></div>
      <div class="rel-stat"><div class="rel-stat-label">Meses</div><div class="rel-stat-val">${meses.length}</div></div>
    </div>
    <p class="section-title">Resumo por mês</p>
    <table><thead><tr><th>Mês/Ano</th><th>Dias</th><th>Total horas</th><th>Média/dia</th></tr></thead>
    <tbody>${mesRows}</tbody></table>`;
}

function exportarCSV(funcId) {
  const f    = funcionarios.find(x => x.id === funcId);
  const regs = registros.filter(r => r.funcId === funcId).sort((a,b)=>a.timestamp-b.timestamp);
  let csv = 'Data,Tipo,Horario,Manual\n';
  regs.forEach(r => {
    const h = new Date(r.timestamp).toLocaleTimeString('pt-BR',{hour:'2-digit',minute:'2-digit'});
    csv += `${r.data},${r.tipo},${h},${r.manual?'Sim':'Nao'}\n`;
  });
  const blob = new Blob(['\uFEFF'+csv], {type:'text/csv;charset=utf-8;'});
  const url  = URL.createObjectURL(blob);
  const a    = document.createElement('a');
  a.href=url; a.download=`ponto_${f.nome.replace(/ /g,'_')}.csv`; a.click();
  URL.revokeObjectURL(url);
  toast('CSV exportado!');
}
