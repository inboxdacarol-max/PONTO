📋 Passo a passo

Siga uma única vez. Leva cerca de 5 minutos.

  1. Acesse sheets.google.com e crie uma planilha nova chamada Ponto AMFERR.
  2. Clique em Extensões → Apps Script.
  3. Apague o código existente e cole o código abaixo: 
    const SHEET_NAME_REG = 'registros';
const SHEET_NAME_FUNC = 'funcionarios';

function doOptions(e) {
  return ContentService.createTextOutput('')
    .setMimeType(ContentService.MimeType.TEXT);
}

function doGet(e)  { return handleRequest(e); }
function doPost(e) { return handleRequest(e); }

function handleRequest(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const p  = e.parameter || {};

  let body = {};
  if (p.payload) {
    body = JSON.parse(decodeURIComponent(p.payload));
  } else if (e.postData) {
    body = JSON.parse(e.postData.contents);
  } else {
    body = p;
  }
  const action   = body.action || p.action || '';
  const callback = p.callback || null;

  let result = {};
  try {
    if (action === 'getFuncionarios') {
      result = { data: lerSheet(ss, SHEET_NAME_FUNC) };
    } else if (action === 'getRegistros') {
      result = { data: lerSheet(ss, SHEET_NAME_REG) };
    } else if (action === 'setFuncionarios') {
      gravarSheet(ss, SHEET_NAME_FUNC, body.data);
      result = { ok: true };
    } else if (action === 'setRegistros') {
      gravarSheet(ss, SHEET_NAME_REG, body.data);
      result = { ok: true };
    } else if (action === 'ping') {
      result = { ok: true };
    } else {
      result = { erro: 'Acao desconhecida: ' + action };
    }
  } catch(err) {
    result = { erro: err.message };
  }

  const json = JSON.stringify(result);

  // JSONP para contornar CORS
  if (callback) {
    return ContentService
      .createTextOutput(callback + '(' + json + ')')
      .setMimeType(ContentService.MimeType.JAVASCRIPT);
  }

  // JSON normal com header CORS
  return ContentService
    .createTextOutput(json)
    .setMimeType(ContentService.MimeType.JSON);
}

function lerSheet(ss, nome) {
  let sheet = ss.getSheetByName(nome);
  if (!sheet) return [];
  const data = sheet.getDataRange().getValues();
  if (data.length < 2) return [];
  const headers = data[0];
  return data.slice(1).map(row => {
    const obj = {};
    headers.forEach((h, i) => { obj[h] = row[i]; });
    return obj;
  });
}

function gravarSheet(ss, nome, dados) {
  let sheet = ss.getSheetByName(nome);
  if (!sheet) sheet = ss.insertSheet(nome);
  sheet.clearContents();
  if (!dados || dados.length === 0) return;
  const headers = Object.keys(dados[0]);
  sheet.appendRow(headers);
  dados.forEach(obj => sheet.appendRow(headers.map(h => obj[h] !== undefined ? obj[h] : '')));
}
